Encapsulation 

#user 
{
	#username 
	{
		font-size: 21px;
		text-align: center;
	}

	&:{ color: rgb( 90, 90, 90 ); } // for #user

	#user_details 
	{
		font-size: 15px;
		border: 1px solid rgb( 123, 123, 123 );
		#username() // gets all properties of #username ( it is a mixin )
	}

	#user_options 
	{
		font-size: 15px;
		border: 2px solid rgb( 123, 123, 123 );
		background-color: rgb( 234, 234, 234 );
	}
}

Mixins 

#user 
{
	.border ( @width, @style, @color ) // parametric mixin
	{
		border: @width @style @color;
	}

	@detached-rules: // detached rulsets
	{
		.small() { font-size: 10px; }
		.medium() { font-size: 15px; }
	};

	#user_details 
	{
		.border ( 1px, solid, rgb( 123, 123, 123 ) );
		font-size: 15px;
		@detached-rules();
		.small();
	}
}

Mixin Conditions 

#user 
{
	.small(@a) when ( @a <= 15px ) { font-size: 5px; }

	.small(@a) when ( @a >= 15px ) { font-size: 15px; }

	#user_details 
	{
		width: 21px;
		.small(9); // makes "font-size: 5px;"
	}
}

For 

#user 
{
	.for_size( @count ) when ( @count > 0 )
	{
		font-size: ( @count );
		.for_size( ( @count - 1px ) );
	}

	#user_details { .for_size( 2 ); }
	// makes "font-size: 2px;" and "font-size: 1px;" in #user_details
}
